## Запуск проекта (для винды):
- $ python -m venv venv
- $ source venv/Scripts/activate
- $ python -m pip install --upgrade pip
- $ pip install -r requirements.txt
- $ cd api_yamdb
- $ python manage.py migrate
- $ python manage.py runserver
